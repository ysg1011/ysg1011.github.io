# Linux 内存管理的水位控制

> buddy 分配 page 的时候，进入伙伴系统算法之前通过函数 `zone_watermark_fast()` 来根据水位判断当前内存情况。内存够的话采用伙伴算法分配，不够的话通过 `node_reclaim` 回收内存。

---

## 1. 水位判断

![水位示意](/images/example.png)

| 水位级别 | 条件 | 行为 |
|---------|------|------|
| **MIN** | 空闲页 < min | 非常缺页，直接在进程上下文进行回收（direct reclaim） |
| **LOW** | 空闲页 < low | 唤醒 kswapd 线程回收（默认 min × 125%） |
| **HIGH** | 空闲页 > high | kswapd 线程可以睡眠（默认 min × 150%） |

> 默认情况下 `low = min × 125%`，`high = min × 150%`，可通过 `watermark_scale_factor` 修改。

---

## 2. 水线计算公式

**总结**：水线默认计算 ZONE_DMA 和 ZONE_NORMAL 区中高于 high 水位的总页数

```bash
# min_free_kbytes 计算
min_free_kbytes = max(min_user_free_kbytes, √(managed × 16))
                = max(11584/4=2896, √(3815346×16)=7813)

# MIN 水线
MIN = page_min × zone_managed_pages / lowmem_pages
    = min_free_kbytes  (只有 NORMAL zone 时)

# LOW 水线
LOW = MIN + max(min × 25%, zone_managed_pages × watermark_scale_factor / 10000)
    = 2896 + max(724, 3815346 × 30 / 10000)
    = 14342

# HIGH 水线
HIGH = LOW + max(min × 25%, zone_managed_pages × watermark_scale_factor / 10000)
     = 14342 + 11446
     = 25788

# PROMO 水线
PROMO = HIGH + max(min × 25%, zone_managed_pages × watermark_scale_factor / 10000)
      = 25788 + 11446
      = 37234
```

## 3. 水位初始化

### 3.1 初始化函数调用链

```c
// online_pages() 和 offline_pages() 也会调用，重新初始化水线并更新 PCP 限制
int __meminit init_per_zone_wmark_min(void)
{
    calculate_min_free_kbytes();          // 1. 计算 min_free_kbytes
    setup_per_zone_wmarks();              // 2. 设置各水位线
    refresh_zone_stat_thresholds();       // 3. 刷新统计阈值
    setup_per_zone_lowmem_reserve();      // 4. 设置预留内存

#ifdef CONFIG_NUMA
    setup_min_unmapped_ratio();           // 5. NUMA 相关
    setup_min_slab_ratio();               // 6. Slab 相关
#endif

    khugepaged_min_free_kbytes_update();  // 7. 大页内存相关

    return 0;
}
postcore_initcall(init_per_zone_wmark_min)
```

### 3.2 calculate_min_free_kbytes()

计算 `min_free_kbytes` 值，最小 128KB，最大 256MB。，如果小于等于  user_min_free_kbytes(默认-1) 则不更改

    min_free_kbytes = 4 * sqrt(lowmem_kbytes)     //高于 high 水位线的 ZONE_DMA 和 ZONE_NORMAL 内存的平方根 的四倍
    
    min_free_kbytes = sqrt(lowmem_kbytes * 16)    // 更好的计算方式，减少误差
    
比如：16MB:    512k

    min_free_kbytes = √(16 * 1024 * 16) = 512
    
user_min_free_kbytes : 通过节点 /proc/sys/vm/min_free_kbytes 修改，写入更改 mini_free_kbytes ，sysctl handle中 user_min_free_kbytes = min_free_kbytes

```c
void calculate_min_free_kbytes(void)
{
    unsigned long lowmem_kbytes;
    int new_min_free_kbytes;

    // 所有 zone 的高于 high 水位线的 ZONE_DMA 和 ZONE_NORMAL 内存
    lowmem_kbytes = nr_free_buffer_pages() * (PAGE_SIZE >> 10);

    // 开根号 √(lowmem_kbytes × 16)
    new_min_free_kbytes = int_sqrt(lowmem_kbytes * 16);

    if (new_min_free_kbytes > user_min_free_kbytes)
        min_free_kbytes = clamp(new_min_free_kbytes, 128, 262144);
    else
        pr_warn("min_free_kbytes is not updated to %d because user defined value %d is preferred\n",
                new_min_free_kbytes, user_min_free_kbytes);
}
```

**参考值表**：

| 内存大小 | min_free_kbytes |
|---------|-----------------|
| 16MB    | 512k            |
| 32MB    | 724k            |
| 64MB    | 1024k           |
| 128MB   | 1448k           |
| 256MB   | 2048k           |
| 512MB   | 2896k           |
| 1024MB  | 4096k           |
| 2048MB  | 5792k           |
| 4096MB  | 8192k           |
| 8192MB  | 11584k          |
| 16384MB | 16384k          |

> 可通过 `/proc/sys/vm/min_free_kbytes` 修改

### 3.3 setup_per_zone_wmarks()

通过 `min_free_kbytes` 修改 MIN、LOW、HIGH、PROMO 水线。

```c
static void __setup_per_zone_wmarks(void)
{
    unsigned long pages_min = min_free_kbytes >> (PAGE_SHIFT - 10);
    unsigned long lowmem_pages = 0;
    struct zone *zone;
    unsigned long flags;

    /* 计算非 HIGHMEM 和非 MOVABLE 的总 page 数 */
    for_each_zone(zone) {
        if (!is_highmem(zone) && zone_idx(zone) != ZONE_MOVABLE)
            lowmem_pages += zone_managed_pages(zone);
    }

    for_each_zone(zone) {
        u64 tmp;

        spin_lock_irqsave(&zone->lock, flags);

        // 计算 MIN 水线
        tmp = (u64)pages_min * zone_managed_pages(zone);
        tmp = div64_ul(tmp, lowmem_pages);

        if (is_highmem(zone) || zone_idx(zone) == ZONE_MOVABLE) {
            // HIGHMEM 和 MOVABLE 特殊处理
            unsigned long min_pages;
            min_pages = zone_managed_pages(zone) / 1024;
            min_pages = clamp(min_pages, SWAP_CLUSTER_MAX, 128UL);
            zone->_watermark[WMARK_MIN] = min_pages;
        } else {
            // 低内存区域，按比例分配
            zone->_watermark[WMARK_MIN] = tmp;
        }

        // 计算 LOW、HIGH、PROMO 水线
        // 取 MIN 的 25% 和 内存的千分之一中的最大值
        tmp = max_t(u64, tmp >> 2,
                mult_frac(zone_managed_pages(zone),
                      watermark_scale_factor, 10000));

        zone->watermark_boost = 0;
        zone->_watermark[WMARK_LOW]  = min_wmark_pages(zone) + tmp;   // MIN 的 125%
        zone->_watermark[WMARK_HIGH] = low_wmark_pages(zone) + tmp;   // MIN 的 150%
        zone->_watermark[WMARK_PROMO] = high_wmark_pages(zone) + tmp; // MIN 的 175%

        spin_unlock_irqrestore(&zone->lock, flags);
    }

    /* 更新 totalreserve_pages */
    calculate_totalreserve_pages();
}
```

### 3.4 refresh_zone_stat_thresholds()

刷新每个 zone 的统计阈值：

```c
void refresh_zone_stat_thresholds(void)
{
    struct pglist_data *pgdat;
    struct zone *zone;
    int cpu;
    int threshold;

    // ... 初始化阈值

    for_each_populated_zone(zone) {
        struct pglist_data *pgdat = zone->zone_pgdat;
        unsigned long max_drift, tolerate_drift;

        threshold = calculate_normal_threshold(zone);

        for_each_online_cpu(cpu) {
            per_cpu_ptr(zone->per_cpu_zonestats, cpu)->stat_threshold = threshold;
            // ...
        }

        tolerate_drift = low_wmark_pages(zone) - min_wmark_pages(zone);
        max_drift = num_online_cpus() * threshold;
        if (max_drift > tolerate_drift)
            zone->percpu_drift_mark = high_wmark_pages(zone) + max_drift;
    }
}
```

### 3.5 setup_per_zone_lowmem_reserve()

计算每个 zone 的预留内存，保证分配内存后还有剩余。

```c
static void setup_per_zone_lowmem_reserve(void)
{
    struct pglist_data *pgdat;
    enum zone_type i, j;

    for_each_online_pgdat(pgdat) {
        for (i = 0; i < MAX_NR_ZONES - 1; i++) {
            struct zone *zone = &pgdat->node_zones[i];
            int ratio = sysctl_lowmem_reserve_ratio[i];
            bool clear = !ratio || !zone_managed_pages(zone);
            unsigned long managed_pages = 0;

            for (j = i + 1; j < MAX_NR_ZONES; j++) {
                struct zone *upper_zone = &pgdat->node_zones[j];
                bool empty = !zone_managed_pages(upper_zone);

                managed_pages += zone_managed_pages(upper_zone);

                if (clear || empty)
                    zone->lowmem_reserve[j] = 0;
                else
                    zone->lowmem_reserve[j] = managed_pages / ratio;
            }
        }
    }

    calculate_totalreserve_pages();
}
```

**相关概念**：
- `spanned`：当前 zone 包含的所有 pages
- `present`：去掉第一阶段 kernel reserve 后的 pages
- `managed`：初始化完成后去掉所有 kernel reserve 后的 pages

> 可通过 `/proc/sys/vm/lowmem_reserve_ratio` 修改

### 3.6 其他初始化函数

```c
// NUMA 系统启用
static void setup_min_unmapped_ratio(void)
{
    pg_data_t *pgdat;
    struct zone *zone;

    for_each_online_pgdat(pgdat)
        pgdat->min_unmapped_pages = 0;

    for_each_zone(zone)
        zone->zone_pgdat->min_unmapped_pages +=
            (zone_managed_pages(zone) * sysctl_min_unmapped_ratio) / 100;
}

// Slab 分配器
static void setup_min_slab_ratio(void)
{
    pg_data_t *pgdat;
    struct zone *zone;

    for_each_online_pgdat(pgdat)
        pgdat->min_slab_pages = 0;

    for_each_zone(zone)
        zone->zone_pgdat->min_slab_pages +=
            (zone_managed_pages(zone) * sysctl_min_slab_ratio) / 100;
}
```

## 4. 相关参数

| 参数 | 路径 | 说明 |
|-----|------|------|
| min_free_kbytes | `/proc/sys/vm/min_free_kbytes` | 最小空闲内存 |
| watermark_scale_factor | `/proc/sys/vm/watermark_scale_factor` | 水线缩放因子（默认 10，即千分之一） |
| lowmem_reserve_ratio | `/proc/sys/vm/lowmem_reserve_ratio` | 低内存预留比例 |
